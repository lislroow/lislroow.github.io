deploy:deploy-file

명령어

set JAVA_HOME=Z:\develop\java\openjdk-11.0.2
set M2_HOME=Z:\develop\build\maven-3.6.3
set PATH=%JAVA_HOME%\bin;%M2_HOME%\bin;%PATH%
set NEXUS_REPO=-DrepositoryId=nexus -Durl=http://nexus/repository/maven-public-hosted/
set GROUP_ID=net.mgkim
set ARTIFACT_ID=spring-core
set VERSION=1.0.0
set FILE=Z:\mgkim-spring-core-1.0.0.jar

%M2_HOME%\bin\mvn.bat -U deploy:deploy-file -DgroupId=%GROUP_ID% -DartifactId=%ARTIFACT_ID% -Dversion=%VERSION% -Dpackaging=jar -Dfile=%FILE% %NEXUS_REPO%
